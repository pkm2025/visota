```python
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

        db_table = 'hr_labor_contract'

        related_name='dependents')

    relationship = models.CharField(max_length=20, choices=Relationship.choices)
    birth_date = models.DateField(null=True, blank=True)
    id_card_no = models.CharField(max_length=20, blank=True, default='')
    tax_code = models.CharField(max_length=20, blank=True, default='')
    deduction_amount = models.DecimalField(max_digits=20, decimal_places=4, default=4400000)
    valid_from = models.DateField()
    valid_to = models.DateField(null=True, blank=True)
    registration_status = models.CharField(max_length=20,
        choices=RegistrationStatus.choices, default=RegistrationStatus.PENDING)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        db_table = 'hr_dependent'
        ordering = ['-valid_from']

        return f'{self.employee.code}: {self.full_name} ({self.relationship})'

    def is_active(self):
        from datetime import date
        today = date.today()
        if self.valid_to and self.valid_to < today:
```

```python
class LeaveBalance(models.Model):
    employee = models.ForeignKey('hr.Employee', on_delete=models.CASCADE,
        related_name='leave_balances')
    fiscal_year = models.SmallIntegerField()
    standard_days = models.DecimalField(max_digits=5, decimal_places=2, default=12)
    carried_forward = models.DecimalField(max_digits=5, decimal_places=2, default=0)
    used_days = models.DecimalField(max_digits=5, decimal_places=2, default=0)

    class Meta:
        db_table = 'hr_leave_balance'
        unique_together = [('employee', 'fiscal_year')]

    @property
    def remaining_days(self):
        return self.standard_days + self.carried_forward - self.used_days
```

```python
from decimal import Decimal
from apps.hr.models.insurance import InsuranceContribution, INSURANCE_CAP

RATES = {
    'bhxh_emp': Decimal('0.08'), 'bhyt_emp': Decimal('0.015'), 'bhtn_emp': Decimal('0.01'),
    'bhxh_er': Decimal('0.17'),  # 14% hưu + 3% ốm/thai
    'bhyt_er': Decimal('0.03'), 'bhtn_er': Decimal('0.01'),
    'bhtnld_er': Decimal('0.005'), 'kpcd_er': Decimal('0.02'),
}

class InsuranceService:
    def __init__(self, company):
        self.company = company

    def calculate_monthly(self, employee, period):
        raw = employee.base_salary
        capped = min(raw, Decimal(str(INSURANCE_CAP)))
        ic, _ = InsuranceContribution.objects.update_or_create(
            employee=employee, period=period,
            defaults={
                'company': self.company,
                'salary_base': capped,
                'bhxh_employee': (capped * RATES['bhxh_emp']).quantize(Decimal('1')),
                'bhyt_employee': (capped * RATES['bhyt_emp']).quantize(Decimal('1')),
                'bhtn_employee': (capped * RATES['bhtn_emp']).quantize(Decimal('1')),
                'bhxh_employer': (capped * RATES['bhxh_er']).quantize(Decimal('1')),
                'bhyt_employer': (capped * RATES['bhyt_er']).quantize(Decimal('1')),
                'bhtn_employer': (capped * RATES['bhtn_er']).quantize(Decimal('1')),
                'bhtnld_employer': (capped * RATES['bhtnld_er']).quantize(Decimal('1')),
                'kpcd_employer': (capped * RATES['kpcd_er']).quantize(Decimal('1')),
            },
        )
        ic.total_employee = ic.bhxh_employee + ic.bhyt_employee + ic.bhtn_employee
        ic.total_employer = (ic.bhxh_employer + ic.bhyt_employer +
                             ic.bhtn_employer + ic.bhtnld_employer)
        ic.save()
        return ic
```
